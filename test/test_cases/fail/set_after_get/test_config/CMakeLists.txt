## Test configuration that calls CMCONF_SET after CMCONF_GET
#
# This should generate a fatal error when CMCONF_SET is called after CMCONF_GET
#

IF(NOT DEFINED CMAKE_SCRIPT_MODE_FILE)
    CMAKE_MINIMUM_REQUIRED(VERSION 3.22)
    PROJECT(CMCONF_SET_AFTER_GET_TEST_CONFIG)
ENDIF()

FIND_PACKAGE(CMLIB REQUIRED)

INCLUDE("${CMAKE_CURRENT_LIST_DIR}/../../../../../CMCONF.cmake")

CMCONF_INIT_SYSTEM("TEST")

CMCONF_GET("VARIABLE_A")

CMCONF_SET("NEW_VAR" "new_value")
