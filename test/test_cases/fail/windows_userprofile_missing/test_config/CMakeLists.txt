## Test configuration that unsets USERPROFILE on Windows
#
# This should generate a fatal error when CMCONF_INIT_SYSTEM is called with UNINSTALL=ON
#

IF(NOT DEFINED CMAKE_SCRIPT_MODE_FILE)
    CMAKE_MINIMUM_REQUIRED(VERSION 3.22)
    PROJECT(CMCONF_WINDOWS_USERPROFILE_MISSING_TEST_CONFIG)
ENDIF()

FIND_PACKAGE(CMLIB REQUIRED)

INCLUDE("${CMAKE_CURRENT_LIST_DIR}/../../../../../CMCONF.cmake")

SET(ENV{USERPROFILE} "")
SET(CMCONF_UNINSTALL ON CACHE BOOL "" FORCE)    
CMCONF_INIT_SYSTEM("TEST")
