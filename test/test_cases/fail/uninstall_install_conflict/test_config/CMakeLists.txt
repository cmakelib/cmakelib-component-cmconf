## Test configuration that sets both UNINSTALL and INSTALL_AS_SYMLINK to ON
#
# This should generate a fatal error when CMCONF_INIT_SYSTEM is called
#

IF(NOT DEFINED CMAKE_SCRIPT_MODE_FILE)
    CMAKE_MINIMUM_REQUIRED(VERSION 3.22)
    PROJECT(CMCONF_UNINSTALL_INSTALL_CONFLICT_TEST_CONFIG)
ENDIF()

FIND_PACKAGE(CMLIB REQUIRED)

INCLUDE("${CMAKE_CURRENT_LIST_DIR}/../../../../../CMCONF.cmake")

SET(CMCONF_UNINSTALL          ON CACHE BOOL "" FORCE)
SET(CMCONF_INSTALL_AS_SYMLINK ON CACHE BOOL "" FORCE)

CMCONF_INIT_SYSTEM("TEST")
